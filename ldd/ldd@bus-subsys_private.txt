struct subsys_private {                          
    struct kset subsys;                          subsys是kset类型，代表bus在sysfs中的类型
    struct kset *devices_kset;                   devices_kset代表bus目录下的device的子目录
                                                 
    struct kset *drivers_kset;                   driver_kset代表bus目录下的driver的子目录
    struct klist klist_devices;                  klist_devices是bus的设备链表，
    struct klist klist_drivers;                  klist_drivers是bus的驱动链表
    struct blocking_notifier_head bus_notifier;  
    unsigned int drivers_autoprobe:1;            
    struct bus_type *bus;                        
                                                 
    struct list_head class_interfaces;           
    struct kset glue_dirs;                       
    struct mutex class_mutex;                    
    struct class *class;                         
};
