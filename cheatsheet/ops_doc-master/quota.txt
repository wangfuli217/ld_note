[root@localhost /]# mount -t ext4 -o usrquota,grpquota /dev/sdb /mnt    #挂载并加入Quota对用户及组限制的参数
[root@localhost /]# cat >> /etc/fstab <<eof                             #对需限制的设备加入{usr,grp}quota参数
> /dev/sdb    /mnt    ext4    default,usrquota,grpquota 0 0
> eof
